 90function accuracy = lab5(testData, testLabels,finalB1L1, finalB1L2, finalW1L1,finalW1L2, finalSoftmaxTheta)%#codegen
 91solved = zeros(10000,1);
 92for x = 1:10000
 93    
 94        %img = func(x, testData);
 95        img = testData(:, x);  %instead of calling the func.m, u can basically just get img directly 
 96
 97%         img = reshape(img,784,1);

 99%         Z2 = {};
100%         Z3 = {};

102        w1 = finalW1L1(:,:);
103        w2 = finalW1L2(:,:);
104        w3 = finalSoftmaxTheta(:,:);
105        b1 = finalB1L1(:,:);
106        b2 = finalB1L2(:,:);
107
108        Z1 = w1*img;
109        Z1 = Z1 + b1;
110        nZ1 = Z1 * -1;
111        for i = 1:200
112            Z1(i, 1) = 1/(1+exp(nZ1(i, 1)));
113        end
114
115        Z2 = w2*Z1;
116        Z2 = Z2 + b2;
117        nZ2 = Z2 * -1;
118        for i = 1:200
119            Z2(i, 1) = 1/(1+exp(nZ2(i, 1)));
120        end
121
122        Z3 = w3*Z2;
123
124        index = 0;
125        max = -1000000;
126        for i = 1:10
127            if Z3(i, 1) > max
128                max = Z3(i, 1);
129                index = i;
130            end
131        end
132        solved(x) = index;
133        
134end

136correct = 0;
137
138for i = 1:10000
139    if testLabels(i, 1) == solved(i,1)

142end
